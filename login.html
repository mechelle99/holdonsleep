// 新增：忘記密碼功能 document.getElementById("btnForgot").onclick = async function() { const uid = prompt("請輸入您的員工編號 (Employee ID)："); if(!uid) return; if(!ENDPOINT) return alert("尚未設定 config.js"); const msg = document.getElementById("msg"); msg.style.display = "block"; msg.textContent = "⏳ 正在查詢並寄送郵件..."; msg.style.color = "#666"; // 暫時改灰色 try { const res = await fetch(ENDPOINT, { method: "POST", headers: { "Content-Type": "text/plain;charset=utf-8" }, body: JSON.stringify({ action: "forgot_pass", data: { id: uid } }) }); const json = await res.json(); if (json.ok) { alert("✅ " + json.message); msg.style.display = "none"; } else { alert("❌ " + json.message); msg.textContent = "❌ " + json.message; msg.style.color = "#c22"; // 錯誤改回紅色 } } catch (e) { alert("❌ 連線錯誤：" + e); msg.style.display = "none"; } };
